units          metal
boundary        p p p
box tilt large
atom_style      atomic
read_data       input.data
pair_style      tersoff
pair_coeff      * * potential.pot A B
fix             int all box/relax iso 0.0 vmax 0.001
compute         stpa all stress/atom NULL

variable        stress_yy equal c_stgb[2]
variable        stress_zz equal c_stgb[3]
variable        stress_xy equal c_stgb[4]
variable        stress_xz equal c_stgb[5]
variable        stress_yz equal c_stgb[6]
thermo_style custom step temp press v_pr etotal c_stgb[1] c_stgb[2] c_stgb[3] c_stgb[4] c_stgb[5] c_stgb[6]
dump            aiida all custom 1 output.traj element x y z  fx fy fz

variable etotal equal etotal
print "final_variable: etotal = ${etotal}"
variable final_energy equal etotal
print "final_energy: ${final_energy}"
print "final_cell: $(xlo) $(xhi) $(xy) $(ylo) $(yhi) $(xz) $(zlo) $(zhi) $(yz)"
